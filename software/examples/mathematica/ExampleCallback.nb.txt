Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="XYZ"(*Change XYZ to the UID of your Joystick Bricklet*)

(*Create IPConnection and device object*)
ipcon=NETNew["Tinkerforge.IPConnection"]
j=NETNew["Tinkerforge.BrickletJoystick",uid,ipcon]
ipcon@Connect[host,port]

(*Callback function for pressed callback*)
PressedCB[sender_]:=
 Print["Pressed"]
AddEventHandler[j@Pressed,PressedCB]

(*Callback function for released callback*)
ReleasedCB[sender_]:=
 Print["Released"]
AddEventHandler[j@Released,ReleasedCB]

Input["Click OK to exit"]

(*Clean up*)
ipcon@Disconnect[]
ReleaseNETObject[j]
ReleaseNETObject[ipcon]
